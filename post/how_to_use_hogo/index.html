## 先初始化站点
hugo new site HugoBlog（在当前目录下生成一个HugoBlog文件夹）

## clone theme
git clone https://github.com/olOwOlo/hugo-theme-even.git themes/even
cp -af themes/even/exampleSite/config.toml .

#其他主题
git clone https://github.com/alex-shpak/hugo-book themes/book

# 本地预览
hugo server -t even --buildDrafts --bind "0.0.0.0" -b "http://your_domain_or_IP:1313"

浏览器访问 http://your_domain_or_IP:1313

# 写一篇文章
# HugoBlog/content/post/how_to_use_hogo.md created
hugo new post/how_to_use_hogo.md

#!/bin/sh

# 任一步骤执行失败都会终止整个部署过程
set -e

printf "\033[0;32mDeploying updates to GitHub...\033[0m\n"

# 构建静态内容
hugo # if using a theme, replace with `hugo -t <YOURTHEME>`

# 切换到 Public 文件夹
cd public

# 添加更改到 git
git add .

# 提交更改
msg="rebuilding site $(date)"
if [ -n "$*" ]; then
msg="$*"
fi
git commit -m "$msg"

# 推送到远程仓库
git push origin main
